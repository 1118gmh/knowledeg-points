## linux常用命令

```
#history 查看历史命令

【绝对路径and相对路径】
#cd /home/jack   绝对
#cd res          相对（相当于cd ./res）
#pwd 显示实际路径

【显示文件】
#ls         显示当前文件夹下所有文件
#ls /root	显示指定目录下所有文件
#ls -a		显示当前文件夹下所有文件，包含影藏的
#ls -l  	显示当前文件夹下所有文件，长列格式显示（= #ll）

【创建文件夹】
#mkdir a    		在当前目录下创建文件夹
#mkdir b c d 		在当前目录下创建多个文件夹
#mkdir /root/a/a1 	在指定目录下创建文件夹
#mkdir -p aa/bb/cc 	在当前目录下递归创建文件夹
#mkdir -m 711 xyz   创建目录，且指定权限(忽略默认权限)

【创建空文件】
#touch a.txt	创建一个空文件并且修改了文件的各个时间
	-a		修改[读取文件时间]为当前时间
	-m		修改[文件内容改变的时间]为当前时间
	-c		修改[文件状态改变的时间]为当前时间

【删除空文件夹】
#rmdir a		在当前目录下删除一个空文件夹
#rmdir a b d	在当前目录下删除多个空文件夹

【删除文件或文件夹】
#rm a.txt 		删除当前目录下的一个文件
#rm *.txt		删除当前目录下以.txt结尾的文件
#rm -r a		递归删除当前目录下非空文件夹
#rm -rf * 		删除当前目录下所有内容

【拷贝】
#cp ./a/a.txt ./b	将一个文件夹中的一个文件复制到另一个文件夹中
#cp ./a/* ./c		将一个文件夹中的所有内容复制到另一个文件夹中
#cp -r a c			将一个文件夹复制到另一个文件夹中

【剪切】
#mv a b				将一个文件夹剪切到另一个文件夹中
#mv a b/aaa			剪切之后重命名
#mv a aaa			重命名a文件夹为aaa
#mv	a1.txt aaa1.txt	重命名a1文件为aaa1
```
